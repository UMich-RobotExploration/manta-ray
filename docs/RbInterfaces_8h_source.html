18 * @brief Enum to capture all the types of sensors
19 * @details This can later be used to parse outputs easier based on these types

21 * When adding a new SensorType. Ensure you override the default kUnknown when

32/** @brief Provides a string conversion for the enum

35constexpr const char *sensorTypeToString(SensorType type) {
36 switch (type) {
37 case SensorType::kUnknown:
38 return "Unknown";
39 case SensorType::kGroundTruthPose:
40 return "GroundTruthPose";
41 case SensorType::kGroundTruthTwist:
42 return "GroundTruthTwist";
43 case SensorType::kPosOdomXY:
44 return "PosOdomXY";
45 default:
46 return "InvalidSensor";
47 }
48}
