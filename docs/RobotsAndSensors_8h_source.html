82/**
83 * @brief GPS-like position sensor.
84 *
85 * @details Only produces a measurement when the robot is within
86 * `surfaceRangeMeters` of the surface (|z - surfaceDepth| <= surfaceRange).
87 * Noise is anisotropic: z-noise is larger than x/y.
88 */
89class GpsPosition : public SensorI {
90public:
91 GpsPosition(double freqHz, int timeSteps,
92 std::normal_distribution<double> xyNoiseDist,
93 std::normal_distribution<double> zNoiseDist,
94 double surfaceRangeMeters = 0.1, double surfaceDepth = 0.0);
95
96 const std::vector<Eigen::VectorXd> &getSensorData() override;
97 const std::vector<double> &getSensorTimesteps() override;
98
99 void updateSensor(const DynamicsBodies &bodies, double simTime,
100 std::mt19937 &rngEngine) override;
101
102 double getXyNoiseStddev() const { return xyNoiseDist_.stddev(); }
103 double getZNoiseStddev() const { return zNoiseDist_.stddev(); }
104
105private:
106 std::normal_distribution<double> xyNoiseDist_;
107 std::normal_distribution<double> zNoiseDist_;
108 double surfaceRangeMeters_{0.1};
109 double surfaceDepth_{0.0};
110};
