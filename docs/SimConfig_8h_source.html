123inline void from_json(const json &j, SimConfig &c) {
124 c.runName = j.value("run_name", c.runName);
125 j.at("output_dir").get_to(c.outputDir);
126 j.at("env_config_file").get_to(c.envConfigFile);
127 c.rngSeed = j.value("rng_seed", c.rngSeed);
128 c.bellhopMemoryMib = j.value("bellhop_memory_mib", c.bellhopMemoryMib);
129
130 if (j.contains("timing")) {
131 const auto &t = j.at("timing");
132 c.endTimeHours = t.value("end_time_hours", c.endTimeHours);
133 c.physicsDt = t.value("physics_dt", c.physicsDt);
134 c.pingIntervalMin = t.value("ping_interval_min", c.pingIntervalMin);
135 c.boundsCheckIntervalSec =
136 t.value("bounds_check_interval_sec", c.boundsCheckIntervalSec);
137 }
138
139 if (j.contains("acoustics")) {
140 const auto &a = j.at("acoustics");
141 c.tofMode = a.value("tof_mode", c.tofMode);
142 c.debugRangeErrorPct =
143 a.value("debug_range_error_pct", c.debugRangeErrorPct);
144 c.numBeams = a.value("num_beams", c.numBeams);
145 c.maxBeams = a.value("max_beams", c.maxBeams);
146 c.beamSpreadDeg = a.value("beam_spread_deg", c.beamSpreadDeg);
147 }
148
149 if (j.contains("sensors")) {
150 j.at("sensors").get_to(c.sensors);
151 }
152
153 if (j.contains("robots")) {
154 c.robotsJson = j.at("robots");
155 }
156
157 if (j.contains("landmarks")) {
158 c.landmarks = j.at("landmarks").get<std::vector<Eigen::Vector3d>>();
159 }
160
161 if (j.contains("pfg")) {
162 j.at("pfg").get_to(c.pfg);
163 }
164}

166inline SimConfig loadSimConfig(const std::string &path) {
167 std::ifstream f(path);
168 if (!f.is_open()) {
169 throw std::runtime_error("Cannot open sim config file: " + path);
170 }
171 auto j = json::parse(f);
172 return j.get<SimConfig>();
