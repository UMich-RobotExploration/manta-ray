26/**
27 * @brief Owner of all rigid body dynamics values
28 * @details Struct provides access to values and ensures lifetimes
29 * @par Why World?
30 * World exists to house the DynamicBodies and robots together. Robots
31 * store indexes into DynamicsBodies, so this way we can ensure
32 * there are no null references,and we can manage the lifecycle of the dynamics
33 * bodies and robots together
34 *
35 * @par Construction
36 * To construct RbWorld, initialize use default construction
37 * and then add robots and landmarks as you see fit.
38 * The only requirement is createRngEngine must be called before advancing the
39 * world to ensure the rng engine is initialized.
40 */

87 * Access them through the index

89 template <typename RobotType, typename... Args>
90 RobotIdx addRobot(Args &&...args) {
91 // First, add a body to get the index
92 BodyIdx newIdx = detail::addDynamicsBody(dynamicsBodies);
93
94 // Create robot and immediately set its index
95 auto robot = std::make_unique<RobotType>(std::forward<Args>(args)...);
96 robot->setDynamicsIndex(newIdx);
97
98 // Store and return reference
99 robots.push_back(std::move(robot));
100 spdlog::info("Adding robot to world at index {}", newIdx);
101
102 return static_cast<RobotIdx>(newIdx);
103 }
