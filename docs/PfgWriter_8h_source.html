26/**
27 * @name Covariance Serialization
28 * @brief PFG covariance format: column-major lower-triangular unique elements.
29 *
30 * @details A symmetric NxN covariance matrix is serialized as N*(N+1)/2
31 * space-separated values. The ordering is **column-major lower-triangular**:
32 * iterate over columns left-to-right, and within each column emit rows from
33 * the diagonal downward.
34 *
35 * This matches `py_factor_graph.utils.matrix_utils`:
36 * - `get_list_column_major_from_symmetric_matrix()` (writer)
37 * - `get_symmetric_matrix_from_list_column_major()` (reader)
38 *
39 * For a 3x3 matrix with indices (row, col):
40 * @code
41 * c0 = (0,0)
42 * c1 = (1,0) c3 = (1,1)
43 * c2 = (2,0) c4 = (2,1) c5 = (2,2)
44 * @endcode
45 * Emitted order: c0 c1 c2 c3 c4 c5 (6 elements)
46 *
47 * For a 6x6 matrix (pose covariance, tangent-space ordering
48 * [x, y, z, roll, pitch, yaw]):
49 * @code
50 * c0 = (0,0)
51 * c1 = (1,0) c6 = (1,1)
52 * c2 = (2,0) c7 = (2,1) c11 = (2,2)
53 * c3 = (3,0) c8 = (3,1) c12 = (3,2) c15 = (3,3)
54 * c4 = (4,0) c9 = (4,1) c13 = (4,2) c16 = (4,3) c18 = (4,4)
55 * c5 = (5,0) c10 = (5,1) c14 = (5,2) c17 = (5,3) c19 = (5,4) c20 =
56 * (5,5)
57 * @endcode
58 * Emitted order: c0 c1 c2 c3 c4 c5 c6 c7 ... c20 (21 elements)
59 *
60 * Diagonal indices for a diagonal-only 6x6: {0, 6, 11, 15, 18, 20}
61 * Diagonal indices for a diagonal-only 3x3: {0, 3, 5}
62 *
63 * @note For diagonal matrices the column-major lower-triangular and row-major
64 * upper-triangular orderings produce identical output because all off-diagonal
65 * elements are zero.
66 * @{
67 */
68inline constexpr int kCovUniqueElems6x6 = 21; ///< 6x6 symmetric: 6*7/2
69inline constexpr int kCovUniqueElems3x3 = 6; ///< 3x3 symmetric: 3*4/2
70/** @} */

72/**
73 * @brief Configuration for the PFG writer.
74 *
75 * @details Controls which data source is used for odometry edges, and provides
76 * covariance parameters for priors and measurements. Per-robot odometry
77 * covariance is derived automatically from each robot's PositionalXYOdometry
78 * sensor noise model.
79 */
80struct PfgWriterConfig {
81 /// If true, odometry edges use GroundTruthPose (noiseless).
82 /// If false, odometry edges use PositionalXYOdometry sensor data.
83 bool useGroundTruthOdometry{true};
84
85 /// Range measurement variance (scalar, for EDGE_RANGE)
86 double rangeVariance{1.0};
87
88 /// Default pose prior covariance (6x6 symmetric, 21 unique elements).
89 /// Serialized as column-major lower-triangular per PFG spec.
90 /// Applied to first pose of each robot.
91 std::array<double, kCovUniqueElems6x6> defaultPosePriorCov{};
92
93 /// Per-landmark prior covariances (3x3 symmetric, 6 unique elements each).
94 /// If empty, landmark priors are not written.
95 /// If size < num landmarks, remaining landmarks use last entry.
96 std::vector<std::array<double, kCovUniqueElems3x3>> landmarkPriorCovs{};
97
98 /// Rotation variance for odometry edges.
99 /// Translation variance is derived per-robot from PositionalXYOdometry noise.
100 double odomRotationVariance{0.0001 * 0.0001};
101
102 /// Rotation variance for GPS pose priors. Should be very large since GPS
103 /// provides no rotation information. Translation covariance is derived
104 /// per-robot from each GpsPosition sensor's noise model.
105 double gpsRotationVariance{1e6};
106};
