115/**
116 * @brief Validates that json structure has proper keys and subkeys
117 *
118 * @tparam N number of subkey to check for
119 *
120 * @details Ensures the following structure exists
121 *
122 * ```json
123 * {
124 * "source_dir" : "some_path",
125 * "subKey": {
126 * "key1": 1,
127 * "key2": 2
128 * }
129 * }
130 * ```
131 */
132template <std::size_t N>
133std::filesystem::path validateJSON(const nlohmann::json &jsonData,
134 const std::string subKey,
135 const std::array<std::string, N> &keys) {
136 if (!jsonData.contains(kSourceKey)) {
137 auto err = fmt::format("Missing required key: ", kSourceKey);
138 throw std::invalid_argument(err);
139 }
140 if (!jsonData.contains(subKey)) {
141 auto err = fmt::format("Missing required sub-key: ", subKey);
142 throw std::invalid_argument(err);
143 }
144 std::filesystem::path rootPath(jsonData[kSourceKey]);
145 if (!std::filesystem::exists(rootPath)) {
146 auto msg = fmt::format("source_dir from config does not exist {}",
147 rootPath.c_str());
148 throw std::invalid_argument(msg);
149 }
150 for (const auto &key : keys) {
151 if (!jsonData[subKey].contains(key)) {
152 throw std::invalid_argument("Missing required key: " + key);
153 }
154 }
155 return rootPath;
156}
